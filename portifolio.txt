            const activeContentElement = document.getElementById(targetId);
            if (activeContentElement && targetId.startsWith('projeto')) {
                resetProjectSections(targetId);
            }
            window.scrollTo({ top: 0, behavior: 'smooth' }); 
        }

        if (headerLogo) {
            headerLogo.addEventListener('click', (e) => {
                e.preventDefault();
                setActiveTab(headerLogo.dataset.tabTarget);
            });
        }

        internalNavLinks.forEach(link => {
             link.addEventListener('click', (e) => {
                e.preventDefault();
                setActiveTab(link.dataset.tabTarget);
            });
        });
        
        function resetProjectSections(projectId) {
            const projectContainer = document.getElementById(projectId);
            if (!projectContainer) return;

            const sectionTabs = projectContainer.querySelectorAll('.project-section-tab');
            const sectionContents = projectContainer.querySelectorAll('.project-section-content');

            if (sectionTabs.length === 0 || sectionContents.length === 0) return;

            sectionTabs.forEach((tab, index) => {
                tab.classList.remove('active-section-tab');
                if (index === 0) tab.classList.add('active-section-tab');
            });
            sectionContents.forEach((content, index) => {
                content.classList.remove('active');
                if (index === 0) content.classList.add('active');
            });
        }

        function initializeProjectSectionTabs() {
            const allProjectContainers = document.querySelectorAll('.tab-content[id^="projeto"]');
            allProjectContainers.forEach(projectContainer => {
                const sectionTabs = projectContainer.querySelectorAll(`.project-section-tab`);
                const sectionContents = projectContainer.querySelectorAll(`.project-section-content`);
                
                if(sectionTabs.length === 0) return;

                sectionTabs.forEach(tab => {
                    tab.addEventListener('click', (e) => {
                        e.preventDefault();
                        const targetId = tab.dataset.sectionTarget;
                        sectionTabs.forEach(innerTab => innerTab.classList.remove('active-section-tab'));
                        tab.classList.add('active-section-tab');
                        sectionContents.forEach(content => {
                            content.classList.remove('active');
                            if (content.id === targetId) {
                                content.classList.add('active');
                            }
                        });
                    });
                });
            });
        }
        
        document.addEventListener('DOMContentLoaded', () => {
            setActiveTab('homePage'); 
            initializeProjectSectionTabs();
        });

    </script>
